168 def build_extension(self, name, module_name=None) -> str | None: 

169 # XXX: This method, add_to_path and set_source_location can be deleted? 

170 if module_name is None: 

171 module_name = name 

172 source_dir = getattr(self, name.lower()) 

173 if source_dir is None: 

174 return None 

175 orig_dir = os.getcwd() 

176 script_args = ["build"] 

177 command = self.distribution.get_command_obj("build") 

178 if command.compiler is not None: 

179 script_args.append(f"--compiler={command.compiler}") 

180 os.chdir(source_dir) 

181 logger.info("building '%s' extension in '%s'", name, source_dir) 

182 run_setup = resolve_name("distutils.core.run_setup") 

183 distribution = run_setup("setup.py", script_args) 

184 ext_modules = distribution.ext_modules 

185 modules = [m for m in ext_modules if m.name == module_name] 

186 if not modules: 

187 msg = f"no module named '{module_name}' in '{source_dir}'" 

188 raise SetupError(msg) 

189 command = distribution.get_command_obj("build_ext") 

190 command.ensure_finalized() 

191 if command.compiler is None: 

192 command.run() 

193 else: 

194 command.build_extensions() 

195 dir_name = os.path.join(source_dir, command.build_lib) 

196 os.chdir(orig_dir) 

197 if dir_name not in sys.path: 

198 sys.path.insert(0, dir_name) 

199 return os.path.join( 

200 source_dir, 

201 command.build_lib, 

202 command.get_ext_filename(module_name), 

203 )

236 def finalize_options(self) -> None: 

237 build = self.get_finalized_command("build") 

238 # check use of deprecated option 

239 options = build.distribution.get_option_dict("build") 

240 if options.get("build_exe", (None, None)) != (None, None): 

241 msg = ( 

242 "[REMOVED] The use of build command with 'build-exe' " 

243 "option is deprecated.\n\t\t" 

244 "Use build_exe command with 'build-exe' option instead." 

245 ) 

246 raise OptionError(msg) 

247 # check values of build_base and build_exe 

248 self.build_base = build.build_base 

249 if self.build_exe == self.build_base: 

250 msg = "build_exe option cannot be the same as build_base directory" 

251 raise SetupError(msg) 

252 if not self.build_exe: # empty or None 

253 self.build_exe = os.path.join(self.build_base, BUILD_EXE_DIR.name) 

254 

255 # make sure all options of multiple values are lists 

256 for option in self.list_options: 

257 setattr(self, option, normalize_to_list(getattr(self, option))) 

258 

259 # path - accepts os.pathsep to be backwards compatible with CLI 

260 if self.path and isinstance(self.path, str): 

261 self.path = self.path.replace(os.pathsep, ",") 

262 include_path = self.include_path 

263 if include_path: 

264 sys_path = sys.path.copy() 

265 cur_path = normalize_to_list(self.path or sys_path) 

266 known_paths = set(cur_path) 

267 for sitedir in include_path: 

268 site.addsitedir(sitedir, known_paths) 

269 self.path = list(known_paths - set(cur_path)) + cur_path 

270 sys.path = sys.path 

271 

272 # the degree of silencing, set from either the silent or silent-level 

273 # option, as appropriate 

274 self.silent = int(self.silent or self.silent_level or 0) 

275 

276 # compression options 

277 self.no_compress = bool(self.no_compress) 

278 if self.zip_filename: 

279 self.zip_filename = os.path.basename( 

280 os.path.splitext(self.zip_filename)[0] + ".zip" 

281 ) 

282 elif self.no_compress is False: 

283 self.zip_filename = "library.zip" 

284

286 if IS_WINDOWS:

288 self.include_msvcr = True

290 else: 

291 self.include_msvcr = False

293 # optimization level: 0,1,2

334 def set_source_location(self, name, *pathParts) -> None: 

335 env_name = f"{name.upper()}_BASE" 

336 attr_name = name.lower() 

337 source_dir = getattr(self, attr_name) 

338 if source_dir is None: 

339 base_dir = os.environ.get(env_name) 

340 if base_dir is None: 

341 return 

342 source_dir = os.path.join(base_dir, *pathParts) 

343 if os.path.isdir(source_dir): 

344 setattr(self, attr_name, source_dir)
