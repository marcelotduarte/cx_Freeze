48class Structure: 

49 """Abstract base class for structures in native byte order. Concrete 

50 structure and union types must be created by subclassing one of these 

51 types, and at least define a _fields class variable. 

52 """ 

53 

54 def __init__(self, *args) -> None: 

55 if not hasattr(self, "_fields"): 

56 self._fields: list[tuple[str, str]] = [] 

57 for i, (field, _) in enumerate(self._fields): 

58 setattr(self, field, args[i]) 

59 

60 def __str__(self) -> str: 

61 dump = json.dumps(self.as_dict(), indent=2) 

62 return self.__class__.__name__ + ": " + dump 

63 

64 def as_dict(self) -> dict[str, str]: 

65 """Return the field values as dictionary.""" 

66 fields = {} 

67 for fieldname, _ in self._fields: 

68 data = getattr(self, fieldname) 

69 if hasattr(data, "as_dict"): 

70 data = data.as_dict() 

71 elif isinstance(data, bytes): 

72 data = data.decode() 

73 fields[fieldname] = data 

74 return fields 

75 

76 def to_buffer(self) -> bytes: 

77 """Return the field values to a buffer.""" 

78 buffer = b"" 

79 for fieldname, fmt in self._fields: 

80 data = getattr(self, fieldname) 

81 if hasattr(data, "to_buffer"): 

82 data = data.to_buffer() 

83 elif isinstance(data, str): 

84 data = data.encode("utf-16le") 

85 elif isinstance(data, int): 

86 data = pack(fmt, data) 

87 buffer += data 

88 return buffer

111class String(Structure): 

112 """File version resource representation of the data.""" 

113 

114 def __init__( 

115 self, key: str, value: int | str | Structure | None = None 

116 ) -> None: 

117 key = key + "\0" 

118 key_len = len(key) 

119 fields = [ 

120 ("wLength", WORD), 

121 ("wValueLength", WORD), 

122 ("wType", WORD), 

123 ("szKey", WCHAR * key_len), 

124 ] 

125 key_len = calcsize(WCHAR) * key_len 

126 pad_len = (4 - ((calcsize(WORD) * 3 + key_len) & 3)) & 3 

127 if 0 < pad_len < 4: 

128 fields.append(("Padding", f"{pad_len}s")) 

129 value_len = 0 

130 value_type = 1 

131 value_size = 1 

132 if isinstance(value, int): 

133 value_len = calcsize(DWORD) 

134 value_type = 0 

135 fields.append(("Value", DWORD)) 

136 elif isinstance(value, str): 

137 value = value + "\0" 

138 value_len = len(value) 

139 value_size = calcsize(WCHAR) 

140 fields.append(("Value", WCHAR * value_len)) 

141 elif hasattr(value, "wLength"): # instance of String 

142 value_len = value.wLength 

143 fields.append(("Value", type(value))) 

144 elif isinstance(value, Structure): 

145 value_len = 0 

146 for field in value._fields: 

147 value_len += calcsize(field[1]) 

148 value_type = 0 

149 fields.append(("Value", type(value))) 

150 

151 self._fields = fields 

152 self.wValueLength = value_len 

153 self.wType = value_type 

154 self.szKey = key 

155 self.Padding = b"\0" * pad_len 

156 self.Value = value 

157 self.wLength = ( 

158 calcsize(WORD) * 3 + key_len + pad_len + value_size * value_len 

159 ) 

160 self._children = 0 

161 

162 def children(self, value: String) -> None: 

163 """Represents the child String object.""" 

164 pad_len = 4 - (self.wLength & 3) 

165 if 0 < pad_len < 4: 

166 field = f"Padding{self._children}" 

167 self._fields.append((field, f"{pad_len}s")) 

168 setattr(self, field, b"\0" * pad_len) 

169 self.wLength += calcsize(CHAR) * pad_len 

170 field = f"Children{self._children}" 

171 self._fields.append((field, type(value))) 

172 setattr(self, field, value) 

173 self._children += 1 

174 self.wLength += value.wLength

177class VersionInfo: 

178 """Organizes the version information (resource) data of a file.""" 

179 

180 def __init__( 

181 self, 

182 version: str, 

183 internal_name: str | None = None, 

184 original_filename: str | None = None, 

185 comments: str | None = None, 

186 company: str | None = None, 

187 description: str | None = None, 

188 copyright: str | None = None, # noqa: A002 

189 trademarks: str | None = None, 

190 product: str | None = None, 

191 dll: bool | None = None, 

192 debug: bool | None = None, 

193 verbose: bool = True, 

194 ) -> None: 

195 valid_version = Version(version) 

196 parts = list(valid_version.release) 

197 while len(parts) < 4: 

198 parts.append(0) 

199 self.version: str = ".".join(map(str, parts)) 

200 self.valid_version: Version = valid_version 

201 self.internal_name: str | None = internal_name 

202 self.original_filename: str | None = original_filename 

203 # comments length must be limited to 31kb 

204 self.comments: str = comments[:COMMENTS_MAX_LEN] if comments else None 

205 self.company: str | None = company 

206 self.description: str | None = description 

207 self.copyright: str | None = copyright 

208 self.trademarks: str | None = trademarks 

209 self.product: str | None = product 

210 self.dll: bool | None = dll 

211 self.debug: bool | None = debug 

212 self.verbose: bool = verbose

214 def stamp(self, path: str | Path) -> None: 

215 """Stamp a Win32 binary with version information.""" 

216 path = Path(path) 

217 if not path.is_file(): 

218 raise FileNotFoundError(path) 

219 

220 if CX_FREEZE_STAMP == "pywin32": 

221 try: 

222 version_stamp = import_module("win32verstamp").stamp 

223 except ImportError as exc: 

224 msg = "install pywin32 extension first" 

225 raise RuntimeError(msg) from exc 

226 # comments length must be limited to 15kb (uses WORD='h') 

227 self.comments = (self.comments or "")[: COMMENTS_MAX_LEN // 2] 

228 version_stamp(os.fspath(path), self) 

229 return 

230 

231 # internal 

232 string_version_info = self.version_info(path) 

233 if CX_FREEZE_STAMP == "internal": 

234 try: 

235 util = import_module("cx_Freeze.util") 

236 except ImportError as exc: 

237 msg = "cx_Freeze.util extension not found" 

238 raise RuntimeError(msg) from exc 

239 handle = util.BeginUpdateResource(path, 0) 

240 util.UpdateResource( 

241 handle, RT_VERSION, ID_VERSION, string_version_info.to_buffer() 

242 ) 

243 util.EndUpdateResource(handle, 0) 

244 

245 if self.verbose: 

246 print("Stamped:", path)

248 def version_info(self, path: Path) -> String: 

249 """Returns the String version info used to stamp the version.""" 

250 major = self.valid_version.major 

251 minor = self.valid_version.minor 

252 micro = self.valid_version.micro 

253 build = 0 

254 file_flags = 0 

255 if self.debug is None or path.stem.lower().endswith("_d"): 

256 file_flags += 1 

257 if self.valid_version.is_devrelease: 

258 file_flags += 8 

259 build = self.valid_version.dev 

260 elif self.valid_version.is_prerelease: 

261 file_flags += 2 

262 build = self.valid_version.pre[1] 

263 elif self.valid_version.is_postrelease: 

264 file_flags += 0x20 

265 build = self.valid_version.post 

266 elif len(self.valid_version.release) >= 4: 

267 build = self.valid_version.release[3] 

268 

269 # use the data in the order shown in 'pepper' 

270 data = { 

271 "FileDescription": self.description or "", 

272 "FileVersion": self.version, 

273 "InternalName": self.internal_name or path.name, 

274 "CompanyName": self.company or "", 

275 "LegalCopyright": self.copyright or "", 

276 "LegalTrademarks": self.trademarks or "", 

277 "OriginalFilename": self.original_filename or path.name, 

278 "ProductName": self.product or "", 

279 "ProductVersion": str(self.valid_version), 

280 "Comments": self.comments or "", 

281 } 

282 is_dll = self.dll 

283 if is_dll is None: 

284 is_dll = path.suffix.lower() in (".dll", ".pyd") 

285 fixed_file_info = VS_FIXEDFILEINFO( 

286 VS_FFI_SIGNATURE, 

287 VS_FFI_STRUCVERSION, 

288 (major << 16) | minor, 

289 (micro << 16) | build, 

290 (major << 16) | minor, 

291 (micro << 16) | build, 

292 VS_FFI_FILEFLAGSMASK, 

293 file_flags, 

294 VOS_NT_WINDOWS32, 

295 2 if is_dll else 1, # VFT_DLL or VFT_APP 

296 0, 

297 0, 

298 0, 

299 ) 

300 

301 # string table with its children 

302 string_table = String(KEY_STRING_TABLE) 

303 for key, value in data.items(): 

304 string_table.children(String(key, value)) 

305 

306 # create string file info and add string table as child 

307 string_file_info = String(KEY_STRING_FILE_INFO) 

308 string_file_info.children(string_table) 

309 

310 # var file info has a child 

311 var_file_info = String(KEY_VAR_FILE_INFO) 

312 var_file_info.children(String("Translation", 0x04E40409)) # 0x409,1252 

313 

314 # VS_VERSION_INFO is the first key and has two children 

315 string_version_info = String(KEY_VERSION_INFO, fixed_file_info) 

316 string_version_info.children(string_file_info) 

317 string_version_info.children(var_file_info) 

318 

319 return string_version_info
