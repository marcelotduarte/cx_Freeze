18# In a MachO file, need to deal specially with links that use @executable_path, 

19# @loader_path, @rpath 

20# 

21# @executable_path - where ultimate calling executable is 

22# @loader_path - directory of current object 

23# @rpath - list of paths to check 

24# (earlier rpaths have higher priority, i believe) 

25# 

26# Resolving these variables (particularly @rpath) requires tracing through the 

27# sequence linked MachO files leading the the current file, to determine which 

28# directories are included in the current rpath.

106 # first reference that gets recorded.

108 # (or None, if no resolution was determined)

118 if not isMachOFile(self.path): 

119 self.isMachO = False

123 self.isMachO = True

125 self.loadCommands = [ 

126 c for c in self.commands if isinstance(c, MachOLoadCommand)

129 c for c in self.commands if isinstance(c, MachORPathCommand)

134 self.resolveLibraryPaths()

137 # this file. 

138 for raw_path, resolved_path in self.libraryPathResolution.items(): 

139 # the path to use for storing in dictionary 

140 if resolved_path is None: 

141 dict_path = Path(raw_path) 

142 else: 

143 dict_path = resolved_path

233 def resolveLoader(self, path: str) -> Path | None: 

234 """Resolve a path that includes @loader_path. @loader_path represents 

235 the directory in which the DarwinFile is located. 

236 """ 

237 if self.isLoaderPath(path): 

238 return self.path.parent / Path(path).relative_to("@loader_path") 

239 msg = f"resolveLoader() called on bad path: {path}" 

240 raise PlatformError(msg)

242 def resolveExecutable(self, path: str) -> Path: 

243 """@executable_path should resolve to the directory where the original 

244 executable was located. By default, we set that to the directory of 

245 the library, so it would resolve in the same was as if linked from an 

246 executable in the same directory. 

247 """ 

248 # consider making this resolve to the directory of the python 

249 # interpreter? Apparently not a big issue in practice, since the 

250 # code has been like this forever. 

251 if self.isExecutablePath(path): 

252 return self.path.parent / Path(path).relative_to( 

253 "@executable_path/" 

254 ) 

255 msg = f"resolveExecutable() called on bad path: {path}" 

256 raise PlatformError(msg)

258 def resolveRPath(self, path: str) -> Path | None: 

259 for rpath in self.getRPath(): 

260 test_path = rpath / Path(path).relative_to("@rpath") 

261 if isMachOFile(test_path): 

262 return test_path 

263 if not self.strict: 

264 # If not strictly enforcing rpath, return None here, and leave any 

265 # error to .finalizeReferences() instead. 

266 return None 

267 print(f"\nERROR: Problem resolving RPath [{path}] in file:") 

268 self.printFileInformation() 

269 msg = f"resolveRPath() failed to resolve path: {path}" 

270 raise PlatformError(msg)

272 def getRPath(self) -> list[Path]: 

273 """Returns the rpath in effect for this file. Determined by rpath 

274 commands in this file and (recursively) the chain of files that 

275 referenced this file. 

276 """ 

277 if self._rpath is not None: 

278 return self._rpath 

279 raw_paths = [c.rpath for c in self.rpathCommands] 

280 rpath = [] 

281 for raw_path in raw_paths: 

282 test_rp = Path(raw_path) 

283 if test_rp.is_absolute(): 

284 rpath.append(test_rp) 

285 elif self.isLoaderPath(raw_path): 

286 rpath.append(self.resolveLoader(raw_path).resolve()) 

287 elif self.isExecutablePath(raw_path): 

288 rpath.append(self.resolveExecutable(raw_path).resolve()) 

289 rpath = [raw_path for raw_path in rpath if raw_path.exists()] 

290 

291 if self.referencing_file is not None: 

292 rpath = self.referencing_file.getRPath() + rpath 

293 self._rpath = rpath 

294 return rpath

296 def resolvePath(self, path: str) -> Path | None: 

297 """Resolves any @executable_path, @loader_path, and @rpath references 

298 in a path. 

299 """ 

300 if self.isLoaderPath(path): # replace @loader_path 

301 return self.resolveLoader(path) 

302 if self.isExecutablePath(path): # replace @executable_path 

303 return self.resolveExecutable(path) 

304 if self.isRPath(path): # replace @rpath 

305 return self.resolveRPath(path) 

306 test_path = Path(path) 

307 if test_path.is_absolute(): # just use the path, if it is absolute 

308 return test_path 

309 test_path = self.path.parent / path 

310 if isMachOFile(test_path): 

311 return test_path.resolve() 

312 if self.strict: 

313 msg = f"Could not resolve path: {path} from file {self.path}." 

314 raise PlatformError(msg) 

315 print( 

316 f"WARNING: Unable to resolve reference to {path} from " 

317 f"file {self.path}. Frozen application may not " 

318 f"function correctly." 

319 ) 

320 return None

370 def _getMachOCommands(path: Path) -> list[MachOCommand]:

373 """ 

374 shell_command = ("otool", "-l", path) 

375 commands: list[MachOCommand] = [] 

376 current_command_lines = None

378 # split the output into separate load commands 

379 out = subprocess.check_output(shell_command, encoding="utf_8") 

380 for raw_line in out.splitlines(): 

381 line = raw_line.strip() 

382 if line[:12] == "Load command":

384 commands.append(

386 )

388 if current_command_lines is not None: 

389 current_command_lines.append(line)

391 commands.append(MachOCommand.parseLines(current_command_lines))

394 @staticmethod 

395 def parseLines(lines: list[str]) -> MachOCommand: 

396 if len(lines) < 2: 

397 return MachOCommand(lines) 

398 parts = lines[1].split(" ") 

399 if parts[0] != "cmd": 

400 return MachOCommand(lines) 

401 if parts[1] == "LC_LOAD_DYLIB": 

402 return MachOLoadCommand(lines)

404 return MachORPathCommand(lines)

626 def finalizeReferences(self) -> None: 

627 """This function does a final pass through the references for all the 

628 copied DarwinFiles and attempts to clean up any remaining references 

629 that are not already marked as copied. It covers two cases where the 

630 reference might not be marked as copied: 

631 1) Files where _CopyFile was called without copyDependentFiles=True 

632 (in which the information would not have been added to the 

633 references at that time). 

634 2) Files with broken @rpath references. We try to fix that up here by 

635 seeing if the relevant file was located *anywhere* as part of the 

636 freeze process. 

637 """ 

638 copied_file: DarwinFile 

639 reference: MachOReference 

640 for copied_file in self._copied_file_list: 

641 for reference in copied_file.getMachOReferenceList(): 

642 if not reference.is_copied: 

643 if reference.isResolved(): 

644 # if reference is resolve, simply check if the resolved 

645 # path was otherwise copied and lookup the DarwinFile 

646 # object. 

647 target_path = reference.resolved_path.resolve() 

648 if target_path in self._darwin_file_for_source_path: 

649 reference.setTargetFile( 

650 self._darwin_file_for_source_path[target_path] 

651 ) 

652 else: 

653 # if reference is not resolved, look through the copied 

654 # files and try to find a candidate, and use it if 

655 # found. 

656 potential_target = self.findDarwinFileForFilename( 

657 reference.raw_path 

658 ) 

659 if potential_target is None: 

660 # If we cannot find any likely candidate, fail. 

661 if self.strict: 

662 copied_file.printFileInformation() 

663 msg = ( 

664 f"finalizeReferences() failed to resolve" 

665 f" path [{reference.raw_path}] in file " 

666 f"[{copied_file.path}]." 

667 ) 

668 raise PlatformError(msg) 

669 print( 

670 "\nWARNING: Could not resolve dynamic link to " 

671 f"[{reference.raw_path}] in file " 

672 f"[{copied_file.path}], and could " 

673 "not find any likely intended target." 

674 ) 

675 continue 

676 print( 

677 f"WARNING: In file [{copied_file.path}]" 

678 f" guessing that {reference.raw_path} " 

679 f"resolved to {potential_target.path}." 

680 ) 

681 reference.resolved_path = potential_target.path 

682 reference.setTargetFile(potential_target)
