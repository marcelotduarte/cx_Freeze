47def scan_code(code: CodeType) -> Generator: 

48 arguments = [] 

49 names = code.co_names 

50 consts = code.co_consts 

51 for _i, _offset, opc, arg in _unpack_opargs(code.co_code): 

52 # keep track of constants (these are used for importing) 

53 # immediately restart loop so arguments are retained 

54 if opc == LOAD_CONST: 

55 arguments.append(consts[arg]) 

56 continue 

57 

58 # keep track of the name which can be the name of the import func 

59 if opc == LOAD_NAME: 

60 arguments.append(names[arg]) 

61 continue 

62 

63 if PUSH_NULL and opc == PUSH_NULL: 

64 continue # ignore it in Python 3.13+ (exists in 3.11+) 

65 

66 if PRECALL and opc == PRECALL: 

67 continue # ignore it in Python 3.11 

68 

69 if (opc, arg) == (CALL or CALL_FUNCTION, 1) and len(arguments) >= 2: 

70 # Python 3.6-3.10 bytecode of a __import__ call: 

71 # 1 0 LOAD_NAME 0 (__import__) 

72 # 2 LOAD_CONST 0 ('pkgutil') 

73 # 4 CALL_FUNCTION 1 

74 # Python 3.11 bytecode of a __import__ call: 

75 # 1 2 PUSH_NULL 

76 # 4 LOAD_NAME 0 (__import__) 

77 # 6 LOAD_CONST 0 ('pkgutil') 

78 # 8 PRECALL 1 

79 # 12 CALL 1 

80 # Python 3.12 bytecode of a __import__ call: 

81 # 1 2 PUSH_NULL 

82 # 4 LOAD_NAME 0 (__import__) 

83 # 6 LOAD_CONST 0 ('pkgutil') 

84 # 8 CALL 1 

85 # Python 3.13 bytecode of a __import__ call: 

86 # 1 2 LOAD_NAME 0 (__import__) 

87 # 4 PUSH_NULL 

88 # 6 LOAD_CONST 0 ('pkgutil') 

89 # 8 CALL 1 

90 func = arguments[-2] 

91 if func in ("__import__", "import_module"): 

92 name = arguments[-1] 

93 yield func, (name, -1, []) 

94 

95 # import statement: attempt to import module 

96 elif opc == IMPORT_NAME: 

97 name = names[arg] 

98 if len(arguments) >= 2: 

99 relative_import_index, from_list = arguments[-2:] 

100 else: 

101 relative_import_index = -1 

102 from_list = arguments[0] if arguments else None 

103 yield "import", (name, relative_import_index, from_list) 

104 

105 # import * statement: copy all global names 

106 elif IMPORT_STAR and opc == IMPORT_STAR: 

107 # Python 3.9-3.11 

108 yield "star", () 

109 elif CALL_INTRINSIC_1 and (opc, arg) == (CALL_INTRINSIC_1, 2): 

110 # Python 3.12-3.13 

111 yield "star", () 

112 

113 # store operation: track only top level 

114 elif opc in STORE_OPS: 

115 name = names[arg] 

116 yield "store", (name,) 

117 

118 # reset arguments; these are only needed for import statements so 

119 # ignore them in all other cases! 

120 arguments = []
